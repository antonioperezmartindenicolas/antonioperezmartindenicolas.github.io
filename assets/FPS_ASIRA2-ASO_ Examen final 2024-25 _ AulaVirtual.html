    function Help {
        Write-Output "Uso: ./csv.ps1 [opción]"
        Write-Output "  -help                   : muestra la ayuda"
        Write-Output "  -crear_csv <archivo>    : Crea un archivo llamado <archivo>"
        Write-Output "  -read_csv <archivo>     : Lee el <archivo>"
        }

    $nombres = @("Juan", "Ana", "Carlos", "María", "Pedro", "Laura", "Miguel", "Elena")
    $apellidos = @("Pérez", "Gómez", "López", "Rodríguez", "Fernández", "Sánchez", "Ramírez", "Torres")
    $Grupos = @("A","B","C")
    $N=20

    function Crear($archivo) {
      if ($archivo -eq $null){
        Write-Output "El archivo no existe"
      }else{ 
        echo "Creamos el archivo $archivo"
        Write-Output "ID,Nombre,apellido, grupo" > $archivo
        for($i=1001; $i -lt $(1001+$N); $i++){
          $nombre=$nombres[$(Get-Random -Maximum $nombres.Length )]
          $apellido=$apellidos[$(Get-Random -Maximum $apellidos.Length)]
          $grupo=$Grupos[$(Get-Random -Maximum $Grupos.Length)]
          $linea="$i,$nombre,$apellido,$grupo"
          Write-Output $linea >> $archivo
        }
      }
    }

    function Read($archivo) {
      if ($archivo -eq $null){
        Write-Output "El archivo no existe"
      }else{ 
        $lista = Import-Csv -Path $archivo
        foreach ($g in $Grupos){
          $n=$($lista |   Where-Object { $_.grupo -eq $g }).Count
          Write-Output "En el grupo $g tenemos $n usuarios"
          foreach ($i in  $lista){
            if($g -eq $i.grupo){
              $a=$i.nombre+" "+$i.apellido+" "+$i.grupo
              echo $a
            }
          }
        }
      }
    }


    if($args.Length -eq 0) {Help}
    if($args.Length -eq 1) { 
      if( $args[0] -eq "-help") {Help}
      }

    foreach ($i in $args){
      if($i -eq "-crear_csv") {Crear $args[1] }
      if($i -eq "-read_csv") {Read $args[1]}
       }

    #!/bin/bash

    mostrar_ayuda() {
    echo "
     -help muestra esta ayuda
     -crear_csv <archivo>
     -read_csv <archivo>
    "
    }
    crear_csv() {
    N=20

    nombres=("Juan" "Ana" "Carlos" "María" "Pedro" "Laura" "Miguel" "Elena")
    longitud_nombres=${#nombres[@]}

    apellidos=("Pérez" "Gómez" "López" "Rodríguez" "Fernández" "Sánchez" "Ramírez" "Torres")
    longitud_apellidos=${#apellidos[@]}

    grupos=("A" "B" "C")
    longitud_grupos=${#grupos[@]}

    echo "Creamos el archivo $1"
    echo "ID,Nombre,apellido, grupo" > $1

    for((i=1001;i<$((1001+N));i++))
    do
      echo $i,${nombres[$((RANDOM%longitud_nombres))]},${apellidos[$((RANDOM%longitud_apellidos))]},${grupos[$((RANDOM%longitud_grupos))]} >> $1

    done

    }

    read_csv() {

    grupos=("A" "B" "C")
    longitud_grupos=${#grupos[@]}
    for((i=0;i<$longitud_grupos;i++))
    do
      echo "En el grupo ${grupos[$i]} tenemos $(sed 's/,/ /' $1 | sed 's/,/ /' | grep -c ",${grupos[$i]}") usuarios: "
      sed 's/,/ /' $1 | sed 's/,/ /' | grep ",${grupos[$i]}"| cut -d',' -f1 | cut -d' ' -f2-
      echo ""
    done
    }


    if [[ $# -eq 0 ]]
    then
      mostrar_ayuda
    elif [[ $1 == "--help" ]]
    then
      mostrar_ayuda
    elif [[ $1 == "--crear_csv" ]]
    then
      if [[ $# > 1 ]]
      then
        crear_csv $2
      else
       echo  "El <archivo> no existe"
      fi
    elif [[ $1 == "--read_csv" ]]
    then
      if [[ $# > 1 ]]
      then
        read_csv $2
      else
       echo  "El <archivo> no existe"
      fi
    else
      mostrar_ayuda
    fi
